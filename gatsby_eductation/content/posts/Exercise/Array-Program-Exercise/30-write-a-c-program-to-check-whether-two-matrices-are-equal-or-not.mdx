int main()
{
   int arr1[50][50], brr1[50][50],i, j, r1, c1, r2, c2, a =1;

     printf("Input elements in the first matrix :\n");
       for(i=0;i<r1;i++)
        {
            for(j=0;j<c1;j++)
            {
               printf("element - [%d],[%d] : ",i,j);

            }
        }

       for(i=0;i<r2;i++)
        {
            for(j=0;j<c2;j++)
            {

            }
        }
      printf("The first matrix is :\n");
     for(i=0;i<r1;i++)
     {
       for(j=0;j<c1 ;j++)
         printf(" %d ",arr1[i][j]);
        printf("\n");
     }
     printf("The second matrix is :\n");
     for(i=0;i<r2;i++)
     {
       for(j=0;j<c2 ;j++)
         printf(" %d ",brr1[i][j]);
        printf("\n");
     }
   if(r1 == r2 && c1 == c2)
   {
        printf("The Matrices can be compared. \n");

          }
