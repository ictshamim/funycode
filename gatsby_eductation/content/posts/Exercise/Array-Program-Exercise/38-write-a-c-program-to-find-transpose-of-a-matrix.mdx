## C Implementation
### 38. Write a C program to find transpose of a matrix.
```c
//Write a C program to find transpose of a matrix.
#include <stdio.h>
void main ()
{
    int array1[3][3], transpose[3][3],i,j,a,b,c,p,q,r;
    printf("Enter 9 co-efficients of the matrix- \n");
    for (i = 0; i < 3; ++i)
    {
        for (j = 0; j < 3; ++j)
        {
            scanf("%d,", &array1[i][j]);
        }
    }
    printf("The given matrix is \n");
    for (i = 0; i < 3; ++i)
    {
        for (j = 0; j < 3; ++j)
            printf(" %d", array1[i][j]);
        printf("\n");
    }
    for (i = 0; i < 3; ++i)
        for (j = 0; j < 3; ++j)
        {
            transpose[j][i] = array1[i][j];
        }
    printf("\nTranspose of the matrix:\n");
    for (i = 0; i < 3; ++i)
        for (j = 0; j < 3; ++j)
        {
            printf("%d  ", transpose[i][j]);
            if (j == 3 - 1)
                printf("\n");
        }
}
```
