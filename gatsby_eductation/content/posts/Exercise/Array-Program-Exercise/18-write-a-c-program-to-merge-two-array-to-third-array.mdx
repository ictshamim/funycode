{
    int arr1[10], arr2[10],arr3[20],i,j,k;

       for(i=0;i<10;i++)

        }

       for(i=0;i<10;i++)
        {
          scanf("%d",&arr2[i]);
        }
     for(i=0;i<10; i++)
        {
            arr3[i] = arr1[i];
         }
     for(j=0;j<10; j++)
        {
            arr3[i] = arr2[j];
            i++;

     for(i=0;i<20; i++)
