## C Implementation
### 26. Write a C program to add two matrices.
```c
//Write a C program to add two matrices.
#include <stdio.h>

void main() {
  int arr1[3][3], brr1[3][3], crr1[3][3], i, j;
  printf("Input elements in the first matrix :\n");
  for (i = 0; i < 3; i++) {
    for (j = 0; j < 3; j++) {
      printf("element - [%d],[%d] : ", i, j);
      scanf("%d", & arr1[i][j]);
    }
  }
  printf("Input elements in the second matrix :\n");
  for (i = 0; i < 3; i++) {
    for (j = 0; j < 3; j++) {
      printf("element - [%d],[%d] : ", i, j);
      scanf("%d", & brr1[i][j]);
    }
  }
  printf("\nThe First matrix is :\n");
  for (i = 0; i < 3; i++) {
    printf("\n");
    for (j = 0; j < 3; j++)
      printf("%d\t", arr1[i][j]);
  }
  printf("\nThe Second matrix is :\n");
  for (i = 0; i < 3; i++) {
    printf("\n");
    for (j = 0; j < 3; j++)
      printf("%d\t", brr1[i][j]);
  }
  for (i = 0; i < 3; i++)
    for (j = 0; j < 3; j++)
      crr1[i][j] = arr1[i][j] + brr1[i][j];
  printf("\nThe Addition of two matrix is : \n");
  for (i = 0; i < 3; i++) {
    printf("\n");
    for (j = 0; j < 3; j++)
      printf("%d\t", crr1[i][j]);
  }
}
```
