## C Implementation
### 29. Write a C program to multiply two matrices.
```c
//Write a C program to multiply two matrices.
#include <stdio.h>
void main()
{
    int arr1[3][3],brr1[3][3],crr1[3][3],i,j;
    printf("Input elements in the first matrix :\n");
    for(i=0; i<3; i++)
    {
        for(j=0; j<3; j++)
        {
            printf("element - [%d],[%d] : ",i,j);
            scanf("%d",&arr1[i][j]);
        }
    }

    printf("Input elements in the second matrix :\n");
    for(i=0; i<3; i++)
    {
        for(j=0; j<3; j++)
        {
            printf("element - [%d],[%d] : ",i,j);
            scanf("%d",&brr1[i][j]);
        }
    }
    printf("\nThe First matrix is :\n");
    for(i=0; i<3; i++)
    {
        printf("\n");
        for(j=0; j<3; j++)
            printf("%d\t",arr1[i][j]);
    }

    printf("\nThe Second matrix is :\n");
    for(i=0; i<3; i++)
    {
        printf("\n");
        for(j=0; j<3; j++)
            printf("%d\t",brr1[i][j]);
    }
    for(i=0; i<3; i++)
        for(j=0; j<3; j++)
            crr1[i][j]=arr1[i][j]*brr1[i][j];
    printf("\nThe subtraction of two matrix is : \n");
    for(i=0; i<3; i++)
    {
        printf("\n");
        for(j=0; j<3; j++)
            printf("%d\t",crr1[i][j]);
    }
}
```

//Write a Java program to multiply two matrices.
import java.util.Scanner;
public class Funycode {
    public static void main(String[] args) {

          Scanner z=new Scanner(System.in);
          int arr1[][]= new int[3][3],brr1[][] = new int[3][3],crr1[][]= new int[3][3],i,j;
          System.out.println("Input elements in the first matrix :\n");
          for(i=0; i<3; i++)
          {
              for(j=0; j<3; j++)
              {
                  System.out.println("Input elements in the first matrix :\n");
                  arr1[i][j] = z.nextInt();
              }
          }
          System.out.println("Input elements in the second matrix :\n");
          for(i=0; i<3; i++)

          for(i=0; i<3; i++)
          {
              System.out.println("\n");
              for(j=0; j<3; j++)
                  System.out.println("\t"+arr1[i][j]);
          }
          System.out.println("\nThe Second matrix is :\n");

          {
              System.out.println("\n");
              for(j=0; j<3; j++)
                  System.out.println("\t"+brr1[i][j]);

          {
              System.out.printf("\n");

                  System.out.printf("\t"+crr1[i][j]);
          }
      }
}
