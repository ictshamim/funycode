## C Implementation
### 28. Write a C program to perform Scalar matrix multiplication.
```c
//Write a C program to perform Scalar matrix multiplication.
#include <stdio.h>
void main()
{
    int arr[3][3],crr[3][3],i,j,n;
    printf("Input elements in the first matrix :\n");
    for(i=0; i<3; i++)
    {
        for(j=0; j<3; j++)
        {
            printf("element - [%d],[%d] : ",i,j);
            scanf("%d",&arr[i][j]);
        }
    }
    printf("\nThe First matrix is :\n");
    for(i=0; i<3; i++)
    {
        printf("\n");
        for(j=0; j<3; j++)
            printf("%d\t",arr[i][j]);
    }

    printf("\nenter the multiplication value :\n");
    scanf("%d",&n);

    for(i=0; i<3; i++)
        for(j=0; j<3; j++)
            crr[i][j]=n*arr[i][j];
    printf("\nThe subtraction of two matrix is : \n");
    for(i=0; i<3; i++)
    {
        printf("\n");
        for(j=0; j<3; j++)
            printf("%d\t",crr[i][j]);
    }
}
```
