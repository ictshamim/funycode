## C Implementation
### 33. Write a C program to find sum of each row and column of a matrix.
```c
//Write a C program to find sum of each row and column of a matrix.
#include <stdio.h>
void main ()
{
    int array[3][3],i,j,sum=0;
    printf("Enter 9 co-efficients of the matrix-\n");
    for (i = 0; i < 3; ++i)
    {
        for (j = 0; j < 3; ++j)
        {
            scanf("%d", &array[i][j]);
        }
    }
    printf("The given matrix is \n");
    for (i = 0; i < 3; ++i)
    {
        for (j = 0; j < 3; ++j)
        {
            printf(" %d", array[i][j]);
        }
        printf("\n");
    }

    for (i = 0; i < 3; ++i)
    {
        for (j = 0; j < 3; ++j)
        {
            sum = sum + array[i][j] ;
        }

        printf("Sum of the %d row is = %d\n", i, sum);
        sum = 0;

    }
    sum = 0;
    for (j = 0; j < 3; ++j)
    {
        for (i = 0; i < 3; ++i)
        {
            sum = sum + array[i][j];
        }

        printf("Sum of the %d column is = %d\n", j, sum);
        sum = 0;

    }

}
```
