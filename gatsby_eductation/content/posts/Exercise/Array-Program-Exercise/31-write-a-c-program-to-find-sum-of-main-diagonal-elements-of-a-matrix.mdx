## C Implementation
### 31. Write a C program to sum of main diagonal elements of a matrix.
```c
//Write a C program to sum of main diagonal elements of a matrix.
#include <stdio.h>
void main()
{
    int arr[3][3],i,j,sum=0;
    printf("Input elements in the matrix :\n");
    for(i=0; i<3; i++)
    {
        for(j=0; j<3; j++)
        {
            printf("element - [%d],[%d] : ",i,j);
            scanf("%d",&arr[i][j]);
        }
    }

    printf("\nThe matrix is : \n");
    for(i=0; i<3; i++)
    {
        printf("\n");
        for(j=0; j<3; j++)
            printf("%d\t",arr[i][j]);
    }
    printf("\n Major diagonal elements are: ");
    for(i=0; i<3; i++)
    {
        for(j=0; j<3; j++)
        {
            if(i==j)
            {
                printf(" %d ",arr[i][j]);
                sum=sum+arr[i][j];
            }
        }
    }
    printf("\nsum of major diagonal element:%d",sum);
}
```
