## C++ Implementation
### 4. Write a C++ program to sort elements of array in ascending order.
```c
//Write a C++ program to sort elements of array in ascending order.


#include<iostream>
using namespace std;
int main()
{
    int arr[10], i, j, temp;
    cout<<"Input 10 elements in the array :\n";
    for(i=0; i<10; i++)
    {
        cin>>arr[i];
    }
    for(i=0; i<10; i++)
    {
        for(j=i+1; j<10; j++)
        {
            if(arr[j] <arr[i])
            {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] =temp;
            }
        }
    }
    cout<<"\nElements of array in sorted ascending order:\n";
    for(i=0; i<10; i++)
    {
        cout<<arr[i]<<" ";
    }
}
```
