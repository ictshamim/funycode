## C Implementation
### 17. Write a C program to count frequency of each element in an array.
```c
//Write a C program to count frequency of each element in an array.
#include <stdio.h>
void main()
{
    int arr[10], fr[10],i,j,ctr;
    printf("Input 10 elements in the array :\n");
       for(i=0;i<10;i++)
        {
          scanf("%d",&arr[i]);
          fr[i] = -1;
        }
    for(i=0; i<10; i++)
    {
        ctr = 1;
        for(j=i+1; j<10; j++)
        {
            if(arr[i]==arr[j])
            {
                ctr++;
                fr[j] = 0;
            }
        }

        if(fr[i]!=0)
        {
            fr[i] = ctr;
        }
    }
    printf("\nThe frequency of all elements of array : \n");
    for(i=0; i<10; i++)
    {
        if(fr[i]!=0)
        {
            printf("%d occurs %d times\n", arr[i], fr[i]);
        }
    }
}
```
