```c
{

      for(j=0;j<3;j++)

      }

 printf("\nThe matrix is : \n");
  for(i=0;i<3;i++)

      printf("\n");
```

## C++ Implementation
### 25. Write a C++ program to enter 3×3 matrix and print it.
```c
//Write a C++ program to enter 3×3 matrix and print it.

#include <iostream>
using namespace std;
int main()
{
    int arr1[50][50], arr2[50][50],i,j,a,b,c,p,q,t,r1,c1,r;
    cout<<"Input Rows and Columns of the  matrix :\n";
    cin>>r1>>c1;
    cout<<"\nInput elements in the  matrix :\n";
    for(i=0; i<r1; i++)
    {
        for(j=0; j<c1; j++)
        {
            cout<<"element -"<<i<<" "<<j<<" : ";
            cin>>arr1[i][j];
            arr2[i][j] = arr1[i][j];
        }
    }
    cout<<"\nThe  matrix is :\n";
    for(i=0; i<r1; i++)
    {
        for(j=0; j<c1 ; j++)
            cout<<arr1[i][j]<<"\t";
        cout<<"\n";
    }
    cout<<"Enter the numbers of two rows to be exchanged \n";
    cin>>a>>b;
    for (i = 0; i < r1; ++i)
    {
        c = arr1[a - 1][i];
        arr1[a - 1][i] = arr1[b - 1][i];
        arr1[b - 1][i] = c;
    }
    cout<<"The matrix after interchanging the two rows(in the original matrix) \n";
    for (i = 0; i < r1; ++i)
    {
        for (j = 0; j < r1; ++j)
        {
            cout<<" "<< arr1[i][j]<<"\t";
        }
        cout<<"\n";
    }
    cout<<"Enter the numbers of two columns to be exchanged \n";
    cin>>p>>q;
    for (i = 0; i < c1; ++i)
    {
        r = arr2[i][p - 1];
        arr2[i][p - 1] = arr2[i][q - 1];
        arr2[i][q - 1] = r;
    }
    cout<<"The matrix after interchanging the two columns(in the original matrix)\n ";
    for (i = 0; i < c1; ++i)
    {
        for (j = 0; j < c1; ++j)
        {
            cout<<"  "<< arr2[i][j]<<"\t";
        }
        cout<<"\n";
    }
}
```
