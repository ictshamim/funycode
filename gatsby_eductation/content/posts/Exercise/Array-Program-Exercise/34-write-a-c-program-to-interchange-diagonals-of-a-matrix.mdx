## C Implementation
### 34. Write a C program to interchange diagonals of a matrix.
```c
//Write a C program to interchange diagonals of a matrix.
#include<stdio.h>

int main()
{
    int array[3][3],i,j,temp;
    printf("Enter 9 co-efficients of the matrix-\n");
    for (i = 0; i < 3; ++i)
    {
        for (j = 0; j < 3; ++j)
        {
            scanf("%d", &array[i][j]);
        }
    }
    printf("The given matrix is \n");
    for (i = 0; i < 3; ++i)
    {
        for (j = 0; j < 3; ++j)
        {
            printf(" %d", array[i][j]);
        }
        printf("\n");
    }
    if(i == j)
    {
        for(i = 0; i < 3; i++)
        {
            temp = array[i][i];
            array[i][i] = array[i][3 - i - 1];
            array[i][3 - i - 1] = temp;
        }

        printf("\n Matrix Elements after Interchanging Diagonals are: \n");
        for(i = 0; i < 3; i++)
        {
            for(j= 0; j <3; j++)
            {
                printf(" %d ", array[i][j]);
            }
            printf("\n");
        }
    }
    else
    {
        printf("\n The Matrix that you entered is Not a Square matrix" );
    }

    return 0;
}
```
