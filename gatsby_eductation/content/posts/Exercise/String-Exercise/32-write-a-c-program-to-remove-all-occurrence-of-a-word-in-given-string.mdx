## C Implementation
### 32. Write a C program to remove all occurrence of a word in given string.
```c
//Write a C program to remove all occurrence of a word in given string.
#include <stdio.h>
#include <string.h>
int main()
{
    char str[100];
    char word[100];
    int i, j, found, ind[10],stringlen, wordlen,p=0;
    printf("Enter any string: ");
    gets(str);
    printf("Enter word to remove: ");
    gets(word);
    stringlen = strlen(str);
    wordlen   = strlen(word);
    ind[p] = -1;
    for(i=0; i<=stringlen - wordlen; i++)
    {
        found = 1;
        for(j=0; j<wordlen; j++)
        {
            if(str[i+j] != word[j])
            {
                found = 0;
                break;
            }
        }
        if(found == 1)
        {
            ind[p++] = i;
        }}
    if(ind[0] == -1)
    {
        printf("'%s' not found.",word);
    }
    else
    {
        for(j=0;j<p;j++){
                ind[j]= ind[j]-j*(wordlen);
        for(i=ind[j]; i <= stringlen - wordlen; i++)
        {
            str[i] = str[i + wordlen];
        }
stringlen= stringlen - wordlen;

        }

        printf("String after removing last '%s': \n%s", word, str);
    }

    return 0;
}
```
