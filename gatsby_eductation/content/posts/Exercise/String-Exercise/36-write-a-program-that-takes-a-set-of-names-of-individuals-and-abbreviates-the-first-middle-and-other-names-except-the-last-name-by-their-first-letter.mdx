//Write a Java program to that takes a set of names of individuals and abbreviates the first, middle and other names except the last name by their first letter.
import java.util.Scanner;

public class Funycode {
    public static void main(String[] args) {
        Scanner z = new Scanner(System.in);
        char[] str = new char[200];
        System.out.println("Input a Full name: ");
        str = z.nextLine().toCharArray();
        int len = str.length;
        int i, p = 0, flag;
        for (i = 0; i < len; i++) {
            if (str[i] == ' ' && str[i + 1] != ' ')
                p++;
        }
        System.out.println("The abbreviated form of the name is: ");
        flag = 0;
        for (i = 0; i < len; i++) {
            if (p > 0) {
                if (i == 0) {
                    System.out.println(str[i]);
                    p--;
                } else if (str[i - 1] == ' ' && str[i] != ' ') {
                    System.out.println(str[i]);
                    p--;
                }
            } else {
                if (flag == 0 && str[i - 1] == ' ')
                    flag = 1;
                if (flag == 1)
                    System.out.print(str[i]);
            }
        }
    }
}
