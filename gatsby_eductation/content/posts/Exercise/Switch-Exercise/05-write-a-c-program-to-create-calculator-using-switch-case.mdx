## C Implementation
### Program-5 Write a C program to create calculator using switch-case.
```c
/*Write a C program to create calculator using switch-case.*/
#include <stdio.h>
int main()
{
char op;
float n1,n2, result=0.0f;
printf("WELCOME TO SIMPLE CALCULATOR\n");
scanf("%f %c %f",&n1,&op,&n2);
  switch(op)
  {
    case '+':
        result = n1+n2;
        break;
    case '-':
        result = n1-n2;
        break;
    case '*':
        result = n1*n2;
        break;
    case '/':
        result = n1/n2;
        break;
    default:
        printf("Invalid operator");
  }
  printf("%.2f %c %.2f = %.2f",n1,op,n2,result);
  return 0;
}
```
### Output
```
Input: 2 + 2
Output: 2+2=4
```

               result = n1-n2;
               break;

               result = n1*n2;
