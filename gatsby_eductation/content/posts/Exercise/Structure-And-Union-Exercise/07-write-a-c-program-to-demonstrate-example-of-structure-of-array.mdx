## C Implementation
### 7. Write a C program to demonstrate example of structure of array.
```c
//Write a C++ program to demonstrate example of structure of array.
#include<stdio.h>
struct Book{
    char booName[100];
    char authorName[100];
};
struct Book book[100];
int count=0;
void input(struct Book *book){
    getchar();
    printf("Book Details \n");
    printf("Book Name : ");
    fgets(book->booName,sizeof(book->booName),stdin);
    printf("author Name : ");
    fgets(book->authorName,sizeof(book->authorName),stdin);
}
void details(struct Book *book){
    printf("Book Details \n\n");
    printf("Book Name   : %s",book->booName);
    printf("author Name : %s",book->authorName);
}
int main(){
        printf("\n\n\nselect from menu: \n");
        printf("1) Add book details.\n2) Display book details.\n3) Exit.\n");
        int k=0;
        scanf("%d",&k);
        system("cls");
        switch(k){
            case 1:{
                if(count>100){
                    printf("memory full");
                    main();
                }
                input(&book[count]);
                count++;
                main();
            }
            case 2:{
                for(int i=0;i<count;i++)
                {
                    details(&book[i]);
                }
                 main();
            }
            case 3:{
             exit(0);
            }
            default: {main();}
        }
}
```
