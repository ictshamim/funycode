## C Implementation
### 16. Write a C program to accept details of ‘n’ employee(eno, ename, salary) and display the details of employee having highest salary. Use array of structure.
```c
//Write a C++ program to accept details of ‘n’ employee(eno, ename, salary) and display the details of employee having highest salary. Use array of structure.
#include<stdio.h>
struct employee{
    char ename[20];
    int eno;
    double salary;
};
void inputdetails(struct employee *a){
    printf("Enter employee details");
    printf("\nName : ");
    fgets(a->ename,sizeof(a->ename),stdin);
    printf("Employee No : ");
    scanf("%d",&a->eno);
    printf("Salary : ");
    scanf("%lf",&a->salary);
    getchar();
}
int main(){
    int n;
    printf("Enter number of employees : ");
    scanf("%d",&n);
    getchar();
    struct employee emp1[n];
    for(int i=0;i<n;i++)
    {
        inputdetails(&emp1[i]);
    }
    struct employee *top;
    int temp=0;
    for(int i=0;i<n;i++)
    {
        if(emp1[i].salary>temp)
        {
            temp=emp1[i].salary;
            top=&emp1[i];
        }
    }
    printf("top paid employee \nname :%semp no: %d\nsalary : %lf",top->ename,top->eno,top->salary);
}
```
