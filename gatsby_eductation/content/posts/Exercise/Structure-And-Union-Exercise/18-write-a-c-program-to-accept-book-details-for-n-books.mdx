## C Implementation
### 18. Write a C program to accept book details for ‘n’ books as book_title, author, publisher and cost. Assign the accession number to each book in increasing order. Display these details as, -Books of a specific author -Books by a specific Publisher -All Books costing Rs. 500 and above. -All Books.
```c
//Write a C program to accept book details for ‘n’ books as book_title, author, publisher and cost. Assign the accession number to each book in increasing order. Display these details as, -Books of a specific author -Books by a specific Publisher -All Books costing Rs. 500 and above. -All Books.
#include<stdio.h>
struct Book{
    char book_title[100];
    char author[100];
    char publisher[50];
    double cost;
    int accessionNumber;
};
int count=0;
struct Book book[1000];
void input(struct Book *book,int i){
    getchar();
    printf("Book Details \n");
    printf("Book Name : ");
    fgets(book->book_title,sizeof(book->book_title),stdin);
    printf("author Name : ");
    fgets(book->author,sizeof(book->author),stdin);
    printf("publisher Name : ");
    fgets(book->publisher,sizeof(book->publisher),stdin);
    book->accessionNumber=i;
    printf("Cost of book : ");
    scanf("%lf",&book->cost);
}
void details(struct Book *book){
    printf("\nBook Details \n\n");
    printf("Book Name      : %s",book->book_title);
    printf("author Name    : %s",book->author);
    printf("publisher Name : %s",book->publisher);
    printf("Price of Book  : %lf",book->cost);
}
int main(){
        if(count==0)
        {
        printf("How many book's do you want to store\n");
        scanf("%d",&count);
        for(int i=0;i<count;i++)
        {
            input(&book[i],i);
        }
        }
        printf("\n\n\nselect from menu: \n");
        printf("1) Books of a specific author.\n2) Books by a specific Publisher.\n3) All Books costing Rs. 500 and above.\n4) All Books\n");
        int k=0;
        scanf("%d",&k);
        system("cls");
        switch(k){
            case 1:{
                int t=0;
                printf("\nEnter author name: ");
                char temp[100];
                getchar();
                fgets(temp,sizeof(temp),stdin);
                for(int i=0;i<count;i++)
                {
                     if(strcmp(book[i].author,temp)==0)
                     {
                         details(&book[i]);
                         t++;
                     }
                }
                t==0? printf("\n NO data found !!") : printf("");
                main();
            }
            case 2:{
                int t=0;
                printf("\nEnter Publisher name: ");
                char temp[100];
                getchar();
                fgets(temp,sizeof(temp),stdin);
                for(int i=0;i<count;i++)
                {
                     if(strcmp(book[i].publisher,temp)==0)
                     {
                         details(&book[i]);
                         t++;
                     }
                }
                t==0? printf("\n NO data found !!") : printf("");
                main();
            }
            case 3:{
                for(int i=0;i<count;i++)
                {
                    if(book[i].cost>500){
                        details(&book[i]) ;
                    }
                }
                main();
            }
            case 4:{
              for(int i=0;i<count;i++)
                {
                        details(&book[i]) ;
                }
                main();
             main();
            }
            default: {main();}
        }
}
```
