## C Implementation
### 17. Write a C program to menu driven program in ‘C’ which shows the working of library. The menu option should be Add book details. Display book details. List all books of given author. List the count of books in the library. Exit.
```c
//Write a C program to menu driven program in ‘C’ which shows the working of library. The menu option should be Add book details. Display book details. List all books of given author. List the count of books in the library. Exit.
#include<stdio.h>
struct Book{
    char booName[100];
    char authorName[100];
};
struct Book book[100];
int count=0;
void input(struct Book *book){
    getchar();
    printf("Book Details \n");
    printf("Book Name : ");
    fgets(book->booName,sizeof(book->booName),stdin);
    printf("author Name : ");
    fgets(book->authorName,sizeof(book->authorName),stdin);
}
void details(struct Book *book){
    printf("Book Details \n\n");
    printf("Book Name   : %s",book->booName);
    printf("author Name : %s",book->authorName);
}
int main(){
        printf("\n\n\nselect from menu: \n");
        printf("1) Add book details.\n2) Display book details.\n3) List all books of given author.\n4) List the count of books in the library.\n5) Exit.\n");
        int k=0;
        scanf("%d",&k);
        system("cls");
        switch(k){
            case 1:{
                if(count>100){
                    printf("memory full");
                    main();
                }
                input(&book[count]);
                count++;
                main();
            }
            case 2:{
                for(int i=0;i<count;i++)
                {
                    details(&book[i]);
                }
                 main();
            }
            case 3:{
            int t=0;
            printf("\nEnter author name: ");
            char temp[100];
            getchar();
            fgets(temp,sizeof(temp),stdin);
            for(int i=0;i<count;i++)
            {
                 if(strcmp(book[i].authorName,temp)==0)
                 {
                     details(&book[i]);
                     t++;
                 }
            }
            t==0? printf("\n NO data found !!") : printf("");
            main();
            }
            case 4:{
             printf("\n\nTotal books in library : %d",count);
             main();
            }
            case 5:{
             exit(0);
            }
            default: {main();}
        }
}
```

2) Display book details.
3) List all books of given author.
4) List the count of books in the library.
