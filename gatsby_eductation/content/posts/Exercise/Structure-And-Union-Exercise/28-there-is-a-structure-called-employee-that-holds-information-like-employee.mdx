//Write a C program There is a structure called employee that holds information like employee code, name, date of joining. printf("emp name : %s\n",emp.name);

    struct Emp emp[3]={"Tushar",12,5,2021,"Imran",11,1,2015,"Arafat",1,1,2010};

    */

    scanf("%d-%d-%d",&day,&month,&year);
    for(int i=0;i<3;i++)
    {
        if(year-emp[i].joinDate.year>3)
        {
            details(emp[i]);
            continue;
        }
         if(year-emp[i].joinDate.year==3){
            if(emp[i].joinDate.month>month)
            {
                details(emp[i]);
                continue;
            }
            if(emp[i].joinDate.month==month)
            {
                if(emp[i].joinDate.day>day)
                {
                    details(emp[i]);
                    continue;
                }
            }
        }
    }
}
