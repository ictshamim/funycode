## C Implementation
### 13. Write a C program to accept batting information of cricket team using structure. It contains player name and runs scored by player. Calculate total runs scored by cricket team.
```c
//Write a C++ program to accept batting information of cricket team using structure. It contains player name and runs scored by player. Calculate total runs scored by cricket team.
#include<stdio.h>
struct player{
    char name[50];
    int score;
};
void input(struct player *a){
    printf("Enter player name : ");
    fgets(a->name,sizeof(a->name),stdin);
    printf("Enter player's score : ");
    scanf("%d",&a->score);
    getchar();
}
int main(){
    struct player *b;
    struct player k[11];
    int sum=0;
    for(int i=0;i<11;i++)
    {
        b=&k[i];
        input(b);
    }
     for(int i=0;i<11;i++)
    {
        sum+=k[i].score;
        printf("%s -- %d \n",k[i].name,k[i].score);
    }
    printf("Total score : ");
    printf("%d",sum);
}
```
