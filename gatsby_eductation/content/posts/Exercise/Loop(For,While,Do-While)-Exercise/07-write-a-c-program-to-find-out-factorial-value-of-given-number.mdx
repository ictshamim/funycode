## C Implementation
### 7. Write a C program to find out factorial value of given number.
```c
//Write a C program to find out factorial value of given number.
// Using for loop
#include<stdio.h>
void main()
{
    int i,n, fact=1;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        fact = fact*i;
    }
    printf("Factorial = %d ",fact);
}
//Using while loop
#include<stdio.h>
void main()
{
    int i=1,n, fact=1;
    scanf("%d",&n);
    while(i<=n)
    {
        fact = fact*i;
        i++;
    }
    printf("Factorial = %d ",fact);
}
// Using do while loop
#include<stdio.h>
void main()
{
    int i=1,n, fact=1;
    scanf("%d",&n);
    do
    {
        fact = fact*i;
        i++;
    }
    while(i<=n);
    printf("Factorial = %d ",fact);
}
```
### Output
```
input 5
Factorial = 120
```
