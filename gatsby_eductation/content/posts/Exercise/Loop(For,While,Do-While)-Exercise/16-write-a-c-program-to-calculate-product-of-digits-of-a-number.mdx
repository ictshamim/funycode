## C Implementation
### 16. Write a C program to calculate product of digits of a number.
```c
//Write a C program to calculate product of digits of a number.
//Using for loop
#include<stdio.h>
void main()
{
    int i,n,pro=1,m;
    printf("Enter a number:");
    scanf("%d",&n);
    for(i=n; n>0; n=n/10)
    {
        m=n%10;
        pro=pro*m;
    }
    printf("product of digits is =%d",pro);
}

//Using while loop
#include<stdio.h>
void main()
{
    int i=1,a,sum=0,avg;
    while(i<=10)
    {
        scanf("%d",&a);
        sum=sum+a;
        i++;
    }
   avg=sum/10;
    printf("sum = %d\n",sum);
    printf("avg = %d",avg);
}
//Using do while loop
#include<stdio.h>
void main()
{
    int i=1,a,sum=0,avg;
    do
    {
        scanf("%d",&a);
        sum=sum+a;
        i++;
    }
    while(i<=10);
    avg=sum/10;
    printf("sum = %d\n",sum);
    printf("avg = %d",avg);
}
```
### Output
```
Enter a number:5
Sum is=5
```
