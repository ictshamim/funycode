## C Implementation
### 36. Write a C program to display the sum of first n terms of Fibonacci series.
```c
//Write a C program to display the sum of first n terms of Fibonacci series.
//Using for loop
#include <stdio.h>
#include <math.h>
int main()
{
      int f1,f2,f3,n,i,s=1;
      f1=0;
      f2=1;
      scanf("%d",&n);
      for(i=2;i<n;i++)
      {
            f3=f1+f2;
            f1=f2;
            f2=f3;
            s=s+f3;
      }
      printf("Sum of %d terms of Fibonacci Series is : %d",n,s);
      return 0;
}
//Using while loop
#include <stdio.h>
#include <math.h>
int main()
{
      int f1,f2,f3,n,i=2,s=1;
      f1=0;
      f2=1;
      scanf("%d",&n);
      while(i<n)
      {
            f3=f1+f2;
            f1=f2;
            f2=f3;
            s=s+f3;
            i++;
      }
      printf("Sum of %d terms of Fibonacci Series is : %d",n,s);
      return 0;
}
//Using do while loop
#include <stdio.h>
#include <math.h>
int main()
{
      int f1,f2,f3,n,i=2,s=1;
      f1=0;
      f2=1;
      scanf("%d",&n);
      do
      {
            f3=f1+f2;
            f1=f2;
            f2=f3;
            s=s+f3;
            i++;
      }
      while(i<n);
      printf("Sum of %d terms of Fibonacci Series is : %d",n,s);
      return 0;
}
```
### Output
```
input 3 4 5
output 12
```
