## C Implementation
### 2. Write a C program to find the sum of first 10 natural numbers.
```c
/*Write a C program to find the sum of first 10 natural numbers.*/
/*Using for loop*/
#include<stdio.h>
void main()
{
   int i,sum=0;
   for(i=1;i<=10;i++)
   {
      sum += i;
   }
   printf("%d ",sum);
}
/*Using while loop*/
#include<stdio.h>
void main()
{
   int i=1,sum=0;
   while(i<=10)
   {
      sum = sum + i;
      i++;
   }
   printf("%d ",sum);
}
/*Using while loop*/
#include<stdio.h>
void main()
{
   int i=1,sum=0;
   do
   {
      sum = sum + i;
      i++;
   }
   while(i<=10);
   printf("%d ",sum);
}
```
### Output
```
45
```
