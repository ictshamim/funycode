## C Implementation
### 5. Write a C program to accept 3 numbers from user one by one and displays their sum on screen.
```c
//Write a C program to accept 3 numbers from user one by one and displays their sum on screen.
//Using for loop
#include<stdio.h>
void main()
{
    int i,a,sum=0;
    for(i=1;i<=3;i++)
    {
        scanf("%d",&a);
        sum=sum+a;
    }
    printf("%d",sum);
}
//Using while loop
#include<stdio.h>
void main()
{
    int i,a,sum=0;
    i=1;
    while(i<=3)
    {
        scanf("%d",&a);
        sum=sum+a;
        i++;
    }
    printf("%d",sum);
}
//Using do while loop
#include<stdio.h>
void main()
{
    int i=1,a,sum=0;
    do
    {
        scanf("%d",&a);
        sum=sum+a;
        i++;
    }
    while(i<=3);
    printf("%d",sum);
}
```
### Output
```
input 3 4 5
output 12
```
