## C Implementation
### 4. Write a C program to print counting from 100 to 1.
```c
//Write a C program to print counting from 100 to 1.
//Using for loop
#include <stdio.h>
void main()
{
    int i;
    for(i=100;i>=1;i--)
    {
        printf("%d ",i);
    }
}
//Using while loop
#include<stdio.h>
void main()
{
    int i=100;
    while(i>=1)
    {
        printf("%d ",i);
        i--;
    }
}
//Using do while loop
#include<stdio.h>
void main()
{
    int i=100;
    do
    {
        printf("%d ",i);
        i--;
    }
     while(i>=1);
}
```
