//Write a Java program in C to display the such a pattern for n number of rows using a number which will start with the number 1 and the first and a last number of each row will be 1.
//Using for loop
import java.util.Scanner;
public class Funycode {
    public static void main(String[] args) {
       Scanner z = new Scanner(System.in);
       int i,j,n;
       System.out.printf("Input number of rows : ");
       n = z.nextInt();
       for(i=0;i<=n;i++)
       {
          for(j=1;j<=n-i;j++)
            System.out.printf(" ");
          for(j=1;j<=i;j++)
            System.out.printf("%d",j);
           for(j=i-1;j>=1;j--)
          System.out.printf("%d",j);
         System.out.printf("\n");
    }
}
