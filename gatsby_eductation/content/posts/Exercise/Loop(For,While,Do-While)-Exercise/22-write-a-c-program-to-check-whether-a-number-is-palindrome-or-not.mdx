## C Implementation
### 22. Write a C program to accept 3 numbers from user one by one and displays their sum on screen.
```c
//Write a C program to accept 3 numbers from user one by one and displays their sum on screen.
//Using for loop
#include<stdio.h>
void main()
{
    int num,r,sum=0,i;
    printf("Input a number: ");
    scanf("%d",&num);
    for(i=num; num!=0; num=num/10)
    {
        r=num % 10;
        sum=sum*10+r;
    }
    if(i==sum)
        printf("%d is a palindrome number.\n",i);
    else
        printf("%d is not a palindrome number.\n",i);
}
//Using while loop
#include<stdio.h>
void main()
{
    int num,r,sum=0,i;
    printf("Input a number: ");
    scanf("%d",&num);
    for(i=num; num!=0; num=num/10)
    {
        r=num % 10;
        sum=sum*10+r;
    }
    if(i==sum)
        printf("%d is a palindrome number.\n",i);
    else
        printf("%d is not a palindrome number.\n",i);
}
//Using do while loop
#include<stdio.h>
void main()
{
    int num,r,sum=0,i;
    printf("Input a number: ");
    scanf("%d",&num);
    i=num;
    do
    {
        r=num % 10;
        sum=sum*10+r;
         num=num/10;
    }
    while(num!=0);
    if(i==sum)
        printf("%d is a palindrome number.\n",i);
    else
        printf("%d is not a palindrome number.\n",i);
}
```
### Output
```
Input a number: 5
5 is a palindrome number.
```
