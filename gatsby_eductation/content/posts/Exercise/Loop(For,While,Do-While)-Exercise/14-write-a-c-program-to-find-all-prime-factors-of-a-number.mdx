## C Implementation
### 14. Write a C program to find all prime factors of a number.
```c
//Write a C program to find all prime factors of a number.
//Using for loop
#include <stdio.h>
int main()
{
    int i, j, num, count;
    scanf("%d", &num);
    printf("All Prime Factors of %d are: \n", num);
    for(i=2; i<=num; i++)
    {
        if(num%i==0)
        {
            count++;
            for(j=2; j<=i/2; j++)
            {
                if(i%j==0)
                {
                     count = 0;
                    break;
                }
            }
            if(count >0)
            {
                printf("%d  ", i);
            }
        }
    }
    return 0;
}
```
