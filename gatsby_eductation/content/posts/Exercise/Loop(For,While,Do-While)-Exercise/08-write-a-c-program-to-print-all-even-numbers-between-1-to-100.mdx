## C Implementation
### 8. Write a C program to print all even numbers between 1 to 100
```c
//Write a C program to print all even numbers between 1 to 100
//Using for loop
#include<stdio.h>
void main()
{
    int i;
    for(i=1;i<=100;i++)
    {
        if(i%2 == 0)
        {
            printf("%d ",i);
        }
    }
}
//Using while loop
#include<stdio.h>
void main()
{
    int i=1;
    while(i<=100)
    {
        if(i%2 == 0)
        {
            printf("%d ",i);
        }
        i++;
    }
}
//Using do while loop
#include<stdio.h>
void main()
{
    int i=1;
    do
    {
        if(i%2 == 0)
        {
            printf("%d ",i);
        }
        i++;
    }

}
```
