## C Implementation
### 20. Write a C program to convert Binary to Hexadecimal number system.
```c
//Write a C program to convert Binary to Hexadecimal number system.
//Using for loop
#include<stdio.h>
int main ()
{
  long int b, hexa = 0, j = 1, r,n;
  printf ("Enter a binary number: ");
  scanf ("%ld", &b);
  for(n=b;b!=0;b = b / 10)
    {
      r = b% 10;
      hexa = hexa+ r * j;
      j = j * 2;
    }
  printf ("Hexadecimal value is: %lX", hexa);
  return 0;
}
//Using while loop
#include<stdio.h>
int main ()
{
  long int b, hexa = 0, j = 1, r,n;
  printf ("Enter a binary number: ");
  scanf ("%ld", &b);
  for(n=b;b!=0;b = b / 10)
    {
      r = b% 10;
      hexa = hexa+ r * j;
      j = j * 2;
    }
  printf ("Hexadecimal value is: %lX", hexa);
  return 0;
}
//Using do while loop
#include<stdio.h>
int main ()
{
    long int b, hexa = 0, j = 1, r;
    printf ("Enter a binary number: ");
    scanf ("%ld", &b);
    do
    {
        r = b% 10;
        hexa = hexa+ r * j;
        j = j * 2;
        b = b / 10;
    }
    while (b!= 0);
    printf ("Hexadecimal value is: %lX", hexa);
    return 0;
}
```
### Output
```
Enter a binary number: 1010
Hexadecimal value is: A
```
