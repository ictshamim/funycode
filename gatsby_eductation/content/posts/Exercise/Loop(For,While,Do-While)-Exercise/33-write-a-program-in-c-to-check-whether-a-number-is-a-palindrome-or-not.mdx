## C Implementation
### 33. Write a C program in C to check whether a number is a palindrome or not.
```c
//Write a C program in C to check whether a number is a palindrome or not.
//Using for loop
#include <stdio.h>

void main(){
    int n,r,sum=0,t;

    printf("Input a number: ");
    scanf("%d",&n);

    for(t=n;n!=0;n=n/10){
         r=n % 10;
         sum=sum*10+r;
    }
    if(t==sum)
         printf("%d is a palindrome number.\n",t);
    else
         printf("%d is not a palindrome number.\n",t);

}
//Using while loop
#include <stdio.h>

void main()
{
    int n,r,sum=0,t;

    printf("Input a number: ");
    scanf("%d",&n);
    t=n;
    while(n!=0)
    {
        r=n % 10;
        sum=sum*10+r;
        n=n/10;
    }
    if(t==sum)
        printf("%d is a palindrome number.\n",t);
    else
        printf("%d is not a palindrome number.\n",t);

}
//Using do while loop
#include <stdio.h>

void main()
{
    int n,r,sum=0,t;

    printf("Input a number: ");
    scanf("%d",&n);
    t=n;
    do
      {
          r=n % 10;
          sum=sum*10+r;
          n=n/10;
      }
        while(n!=0);
    if(t==sum)
        printf("%d is a palindrome number.\n",t);
    else
        printf("%d is not a palindrome number.\n",t);

}
```
### Output
```
Input a number: 5
5 is a palindrome number.
```
