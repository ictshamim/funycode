## C Implementation
### 18. Write a C program to convert Binary to Octal number system.
```c
//Write a C program to convert Binary to Octal number system.
//Using for loop
#include <stdio.h>

void main()
{
    long int b, oct= 0, j = 1, r,n;

    printf("Enter the value for  binary number: ");
    scanf("%ld", &b);
    for (n=b;b!=0;b=b/10)
    {
        r = b % 10;
        oct = oct + r * j;
        j = j * 2;

    }
    printf("Equivalent octal value is: %lo", oct);
}
//Using while loop
#include <stdio.h>

void main()
{
    long int b, oct= 0, j = 1, r;

    printf("Enter the value for  binary number: ");
    scanf("%ld", &b);
    while (b != 0)
    {
        r = b % 10;
        oct = oct + r * j;
        j = j * 2;
        b = b / 10;
    }
    printf("Equivalent octal value: %lo", oct);
}
//Using do while loop
#include <stdio.h>

void main()
{
    long int b, oct= 0, j = 1, r;

    printf("Enter the value for  binary number: ");
    scanf("%ld", &b);
    do
    {
        r = b % 10;
        oct = oct + r * j;
        j = j * 2;
        b = b / 10;
    }
    while (b != 0);
    printf("Equivalent octal value: %lo", oct);
}
```
### Output
```
Enter the value for  binary number: 10101
Equivalent octal value is: 25
```
