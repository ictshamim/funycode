## C Implementation
### 23. Write a C program to find frequency of each digit in a given integer.
```c
//Write a C program to find frequency of each digit in a given integer.
//Using for loop
#include <stdio.h>
void main()
{
    int n, i, j, ctr, r;
    printf("Input any number: ");
    scanf("%d",&n);
    for (i = 0; i < 10; i++)
    {
        printf("The frequency of  %d = ",i);
        ctr = 0;
        for (j = n; j > 0; j = j / 10)
        {
            r = j % 10;
            if (r == i)
            {
                ctr++;
            }
        }
        printf("%d",ctr);
        printf("\n");
    }
}
//Using while loop
#include <stdio.h>
void main()
{
    int n, i, j, ctr, r;
    printf(" Input any number: ");
    scanf("%d",&n);
    i = 0;
    while( i < 10)
    {
        printf("The frequency of  %d = ",i);
        ctr = 0;
        for (j = n; j > 0; j = j / 10)
        {
            r = j % 10;
            if (r == i)
            {
                ctr++;
            }
        }
        printf("%d",ctr);
        printf("\n");
        i++;
    }
}
//Using do while loop

{
    int n, i, j, ctr, r;

    i = 0;

            {
                ctr++;
            }
        }
        printf("%d",ctr);
        printf("\n");
        i++;
    }
    while( i < 10);
}
```
