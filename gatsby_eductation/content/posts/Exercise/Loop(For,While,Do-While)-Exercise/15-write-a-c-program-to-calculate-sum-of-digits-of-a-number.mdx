## C Implementation
### 15. Write a C program to calculate sum of digits of a number.
```c
//Write a C program to calculate sum of digits of a number.
//Using for loop
#include<stdio.h>
void main()
{
    int i,n,sum=0,m;
    printf("Enter a number:");
    scanf("%d",&n);
    for(i=n;n>0;n=n/10)
    {
      m=n%10;
      sum=sum+m;
    }
    printf("Sum is=%d",sum);
}

//Using while loop
#include<stdio.h>
void main()
{
    int i,n,sum=0,m;
    printf("Enter a number:");
    scanf("%d",&n);
    i=n;
    while(n>0)
    {
      m=n%10;
      sum=sum+m;
    }
    n=n/10;
    printf("Sum is=%d",sum);
}
//Using do while loop
#include<stdio.h>
void main()
{
    int n,sum=0,m;
    printf("Enter a number:");
    scanf("%d",&n);
    do
    {
        m=n%10;
        sum=sum+m;
        n=n/10;
    }
    while(n>0);
    printf("Sum is =%d",sum);
}
```
### Output
```
Enter a number:5
Sum is=5
```
