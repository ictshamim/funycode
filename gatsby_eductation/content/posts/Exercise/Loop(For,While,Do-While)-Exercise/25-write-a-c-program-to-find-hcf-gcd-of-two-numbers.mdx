## C Implementation
### 25. Write a C program to find HCF (GCD) of two numbers.
```c
//Write a C program to find HCF (GCD) of two numbers.
//Using for loop
#include <stdio.h>

void main()
{
    int i, n1, n2, j, hcf=1;

    scanf("%d%d",&n1,&n2);

    j = (n1<n2) ? n1 : n2;

    for(i=1; i<=j; i++)
    {

        if(n1%i==0 && n2%i==0)
        {
            hcf = i;
        }
    }

    printf("\nHCF of %d and %d is : %d\n\n", n1, n2, hcf);
}
//Using while loop
#include <stdio.h>

void main()
{
    int i, n1, n2, j, hcf=1;

    scanf("%d%d",&n1,&n2);

    j = (n1<n2) ? n1 : n2;

    i=1;

    while(i<=j)
    {

        if(n1%i==0 && n2%i==0)
        {
            hcf = i;
        }
         i++;
    }

    printf("\nHCF of %d and %d is : %d\n\n", n1, n2, hcf);
}
//Using do while loop
#include <stdio.h>

void main()
{
    int i, n1, n2, j, hcf=1;

    scanf("%d%d",&n1,&n2);

    j = (n1<n2) ? n1 : n2;

    i=1;
    do
    {

        if(n1%i==0 && n2%i==0)
        {
            hcf = i;
        }
         i++;
    }
        while(i<=j);

    printf("\nHCF of %d and %d is : %d\n\n", n1, n2, hcf);
}
```
### Output
```
2 4
HCF of 2 and 4 is : 2
```
