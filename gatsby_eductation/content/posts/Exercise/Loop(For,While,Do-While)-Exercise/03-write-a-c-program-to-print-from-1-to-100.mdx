## C Implementation
### 3. Write a C program to print from 1 to 100.
```c
//Write a C program to print from 1 to 100.
// Using for loop
#include<stdio.h>
void main()
{
    int i;
    for(i=1;i<=100;i++)
    {
        printf("%d ",i);
    }
}
//Using while loop
#include<stdio.h>
void main()
{
    int i=1;
    while(i<=100)
    {
        printf("%d ",i);
        i++;
    }
}
//Using do while
#include<stdio.h>
void main()
{
    int i=1;
    do
    {
        printf("%d ",i);
        i++;
    }
    while(i<=100);
}
```
