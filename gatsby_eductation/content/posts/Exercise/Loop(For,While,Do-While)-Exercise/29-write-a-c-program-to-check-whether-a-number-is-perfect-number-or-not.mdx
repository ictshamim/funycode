## C Implementation
### 29. Write a C program to check whether a number is Perfect number or not.
```c
//Write a C program to check whether a number is Perfect number or not.
//Using for loop
#include<stdio.h>
void main(){
  int n,i,sum=0;

  printf("Enter a number: ");
  scanf("%d",&n);

  for(i=1;i<n;i++){
      if(n%i==0)
           sum=sum+i;
  }
  if(sum==n)
      printf("%d is a perfect number",i);
  else
      printf("%d is not a perfect number",i);

}
//Using while loop
#include<stdio.h>
void main()
{
    int n,i=1,sum=0;

    printf("Enter a number: ");
    scanf("%d",&n);

    while(i<n)
    {
        if(n%i==0)
            sum=sum+i;
        i++;
    }
    if(sum==n)
        printf("%d is a perfect number",i);
    else
        printf("%d is not a perfect number",i);

}
//Using do while loop
#include<stdio.h>
void main()
{
    int n,i=1,sum=0;

    printf("Enter a number: ");
    scanf("%d",&n);

    do
    {
        if(n%i==0)
            sum=sum+i;
        i++;
    }
    while(i<n);
    if(sum==n)
        printf("%d is a perfect number",i);
    else
        printf("%d is not a perfect number",i);

}
```
### Output
```
Enter a number: 5
5 is not a perfect number
```
