//Write a C program to accept 3 numbers from user one by one and displays their sum on screen.
//Using for loop
#include <stdio.h>
int main()
{
    int i,j, n, num = 0;

    printf("Enter any number to print in words: ");
    scanf("%d", &n);

    for(i=n;n!=0; n /= 10)
    {
        num = (num * 10) + (n % 10);
    }

    for(j=num;num!= 0;num = num/10)
    {
        switch(num % 10)
        {
            case 0:
