## C Implementation
### 6. Write a C program to take a number from user and check is it prime number or not.
```c
//Write a C program to take a number from user and check is it prime number or not.
//Using for loop
#include<stdio.h>
void main()
{
    int i,n,count = 0;
    scanf("%d",&n);
    for(i=2;i<n;i++)
    {
        if(n%i == 0)
        {
            count++;
            break;
        }
    }
    if(count == 0)
    {
            printf("%d is a  prime number",n);
    }
    else
    {
            printf("%d is not a  prime number",n);
    }
}

//Using while loop
#include<stdio.h>
void main()
{
    int i=2,n,count = 0;
    scanf("%d",&n);
    while(i<n)
    {
        if(n%i == 0)
        {
            count++;
            break;
        }
        i++;
    }
    if(count == 0)
    {
            printf("%d is a  prime number",n);
    }
    else
    {
            printf("%d is not a  prime number",n);
    }
}
//Using do while loop
#include<stdio.h>
void main()
{
    int i=2,n,count = 0;
    scanf("%d",&n);
    do
    {
        if(n%i == 0)
        {
            count++;
            break;
        }
        i++;
    }
    while(i<n);
    if(count == 0)
    {
            printf("%d is a  prime number",n);
    }
    else
    {
            printf("%d is not a  prime number",n);
    }
}
```
### Output
```
input 6
output 6 is not a  prime number
```
