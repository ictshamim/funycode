## C Implementation
### 7. Write a C program to store n elements in an array and print the elements using pointer.
```c
//Write a C program to store n elements in an array and print the elements using pointer.
#include <stdio.h>
void main()
{
    int a[1000],n,i;
    int * p = a;

    printf("Enter size of array: ");
    scanf("%d", &n);

    printf("Enter elements in array:\n");
    for (i = 0; i < n; i++){
        scanf("%d", p);        //input the element
        p++;
    }
    p = a;
    printf("Array elements: ");
    for (i = 0; i < n; i++){
        printf("%d, ", *p);   //print the element
        p++;
    }
}
```
### Output
```
Enter size of array: 3
Enter elements in array:
1 2 3
Array elements: 1, 2, 3,
```
## C++ Implementation
### 7. Write a C++ program to store n elements in an array and print the elements using pointer.
```c
//Write a C++ program to store n elements in an array and print the elements using pointer.
#include <iostream>
using namespace std;
#include <iostream>
using namespace std;

int main()
{
    int a[1000],n,i;
    int * p = a;

    cout<<("Enter size of array: ");
    cin>>n;

    cout<<("Enter elements in array:\n");
    for (i = 0; i < n; i++){
        cin>>*p;        //input the element
        p++;
    }
    p = a;
    cout<<("Array elements: ");
    for (i = 0; i < n; i++){
        cout<< *p<<" ";   //print the element
        p++;
    }
}
```

