### Percentage >= 40% : Grade E
### Percentage < 40% : Grade F
```c
#include<stdio.h>
void main()
{
    int s;
    char g,q;
       printf("Enter a employee's year's of service, gender & qualification-");
    scanf("%d %c %c",&s,&g,&q);
    if(s>=10&&g=='m'&&q=='p')
        printf("salary 15000");

        printf("salary 10000");
    else if(s<10&&g=='m'&&q=='d')
```

## C++ Implementation
### Program-35: Write a C program to input marks of five subjects Physics, Chemistry, Biology, Mathematics and Computer. Calculate percentage and grade according to following:
### Percentage >= 90% : Grade A
### Percentage >= 80% : Grade B
### Percentage >= 70% : Grade C
### Percentage >= 60% : Grade D
### Percentage >= 40% : Grade E
### Percentage < 40% : Grade F
```c
#include<iostream>
using namespace std;
int main()
{
    int a,p,c,b,m,e,A,B,C,D,E,F;
    cout<<"enter the marks of five subjects Physics, Chemistry, Biology, Mathematics and Computer-\n";
    cin>>p>>c>>b>>m>>e;
    a=((p+c+b+m+e)*100)/500;
    cout<<"average="<<a<<"\n";
    if(a>=90)
        cout<<"grade A";
    else if(a<=89&&a>=80)
        cout<<"grade B";
    else if(a<=79&&a>=70)
        cout<<"grade C";
    else if(a<=69&&a>=60)
        cout<<"grade D";
    else if(a<=59&&a>=40)
        cout<<"grade E";
    else
        cout<<"grade F";
}
```
