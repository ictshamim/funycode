### 1. Write an assembly program to enter one of the hex digits A-F, and display it on the next line in decimal.
```
.model small
 .stack 100h

 .data
    msg_1  equ  \'enterr the hex digit (a-f) : $\'
    msg_2  equ  0dh,0ah,\'the corresponding decimal dgit is : $\'


 .code
   main proc
     mov ax, @DATA
     mov ds, ax

     lea dx, msg_1
     mov ah, 9
     int 21h

     mov ah, 1
     int 21h

     mov bl, al

     lea dx, msg_2
     mov ah, 9
     int 21h

     mov ah, 2
     mov dl, 31h
     int 21h

     sub bl, 11h

     mov dl, bl
     int 21h

     mov ah, 4ch
     int 21h
   main endp
 end main

```

## Input
```
A
```
## Output
```
10
```
