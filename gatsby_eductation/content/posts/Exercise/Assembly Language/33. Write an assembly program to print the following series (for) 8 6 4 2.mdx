### 33. Write an assembly program to print the following series  (for)  8 6 4 2

```
.model small
.stack 100h
.data
mg1 db ?
mg2 db ?
.code
main proc

 mov ax,@data
 mov ds,ax

 mov cx,8
 jcxz skip
 top:
  mov ah,2

  mov bx,cx
  add bx,30h
  mov dx,bx
  sub cx,1
  int 21h
  loop top
  jmp skip

  skip:
 mov ah,4ch
 int 21h

 main endp
 end main
```

## Output
```
8 6 4 2
```
