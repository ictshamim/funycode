### 1.
```
.model small
.stack 100h
.data
    str1 db 0ah,0dh,'the sum of '
    firstnum db ?
    str2 db ' and '
    secondnum db ?
    str3 db ' is '
    ans db ?
    str4 db ' $'
.code
main proc

    mov ax,@data
    mov ds,ax

    mov ah,2
    mov dl,3fh
    int 21h

    mov ah,1
    int 21h
    mov bl,al
    mov firstnum,al
    int 21h
    mov secondnum,al

    add bl,al
    sub bl,30h
    mov ans,bl

    mov ah,9
    lea dx,str1
    int 21h

    mov ah,4ch
    int 21h

    main endp
end main

```

## Input
```
?23
```
## Output
```
the sum of 2 and 3 is 5
```
