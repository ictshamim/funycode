### 31. Write an assembly program to print the following series (for) 9 7 5 3 1

```
.model small
.stack 100h
.data
mg1 db ?
mg2 db ?
.code
main proc

 mov ax,@data
 mov ds,ax

 mov cx,10
 jcxz skip
 top:
  mov ah,2
  sub cx,1
  mov bx,cx
  add bx,30h
  mov dx,bx
  int 21h
  loop top
  jmp skip

  skip:
 mov ah,4ch
 int 21h

 main endp
 end main
```

## Output
```
9 7 5 3 1
```
