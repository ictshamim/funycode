## C Implementation
### 7. Write a C program to check whether a number is a prime number or not using the function.
```c
//Write a C program in C to check whether a number is a prime number or not using the function.
#include<stdio.h>

int square(int num)
{
    return num*num;
}

int main()
{
    int n;
    printf("Enter a number: ");
    scanf("%d",&n);

    int ans=square(n);

    printf("%d\n",ans);
    return 0;
}
```
### Output
```
Enter a number: 4
16
```

## C++ Implementation
### 7. Write a C++ program to check whether a number is a prime number or not using the function.
```c
//Write a C++ program in C to check whether a number is a prime number or not using the function.

#include<iostream>
using namespace std;

void checkPrime(int n)
{
    int check=1;
    for(int i=2;i*i<=n;i++)
    {
        if(n%i==0)
        {
            check=0;
            break;
        }
    }
    if(check && n!=1)
        cout << n << " is prime";
    else
        cout << n << " is not prime";
}
int main()
{
    int n;
    cin >> n;

    checkPrime(n);
    return 0;
}
```
### Output
```
Enter a number: 4
16
```
