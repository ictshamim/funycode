## C Implementation
### 9. Write a C program to find diameter, circumference and area of circle using functions.
```c
//Write a C program to find diameter, circumference and area of circle using functions.
#include<stdio.h>

int square(int num)
{
    return num*num;
}

int main()
{
    int n;
    printf("Enter a number: ");
    scanf("%d",&n);

    int ans=square(n);

    printf("%d\n",ans);
    return 0;
}
```
### Output
```
Enter radius of circle: 3
Diameter is 6
Circumference is 18.85
```

## C++ Implementation
### 9. Write a C++ program to find diameter, circumference and area of circle using functions.
```c
//Write a C++ program to find diameter, circumference and area of circle using functions.

#include<iostream>
using namespace std;
void diameter(int n)
{
    cout << "Diameter is " << n*2 << endl;
}

void circumference(int n)
{
    float pi=3.14159,ans;
    ans=2*pi*n;
    cout << "Circumference is " << ans << endl;
}

int main()
{
    int rad;
    cout << "Enter radius of circle: ";
    cin >> rad;

    diameter(rad);

    circumference(rad);

    return 0;
}
```
### Output
```

Diameter is 6
```
