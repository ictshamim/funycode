## C Implementation
### 22. Write a C programming to find out maximum and minimum of some values using function which will return an array.
```c
//Write a C program to find the square of any number using the function.
#include<stdio.h>

int findMAX(int ara[],int n)
{
    int mx=ara[0];
    for(int i=1;i<n;i++)
    {
        if(mx<ara[i])
            mx=ara[i];
    }
    return mx;
}

int findMIN(int ara[],int n)
{
    int mn=ara[0];
    for(int i=1;i<n;i++)
    {
        if(mn>ara[i])
            mn=ara[i];
    }
    return mn;
}

int main()
{
    int n;
    printf("Enter size of Array: ");
    scanf("%d",&n);
    int ara[n];
    printf("Enter elements of Array: ");
    for(int i=0;i<n;i++)
        scanf("%d",&ara[i]);

    int mx=findMAX(ara,n);
    int mn=findMIN(ara,n);

    printf("Maximum value is: %d\n",mx);
    printf("Minimum value is: %d\n",mn);

    return 0;
}
```
### Output
```
Enter size of Array: 3
Enter elements of Array: 1 2 3
Maximum value is: 3
Minimum value is: 1
```
