## C Implementation
### 41. Write a C program to Calculate the Value of sin(x)
```c
//Write a C program to Calculate the Value of sin(x)
#include<stdio.h>

float Sinx(float x,int n)
{
    int fact,sign=-1,i,j;
    float p,sum=0.0;
    for (i = 1; i <= n; i += 2)
    {
        p = 1;
        fact = 1;
        for (j = 1; j <= i; j++)
        {
            p = p * x;
            fact = fact * j;
        }
        sign =  (- 1) * sign;
        sum += sign * p / fact;
    }
    return sum;
}

int main()
{
    float x;
    int n;
    printf("Enter value of x: ");
    scanf("%f",&x);
    printf("Enter value of n: ");
    scanf("%d",&n);

    float val=Sinx(x,n);

    printf("sin(%0.2f) = %.2f", x, val);
    return 0;
}
```

```

Enter value of n: 4
sin(3.00) = -1.50
```
