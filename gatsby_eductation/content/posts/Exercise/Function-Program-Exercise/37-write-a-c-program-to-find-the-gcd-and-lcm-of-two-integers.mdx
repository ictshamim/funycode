## C Implementation
### 37. Write a C program to Find the GCD and LCM of Two Integers
```c
//Write a C program to Find the GCD and LCM of Two Integers
#include<stdio.h>

int findGCD(int a,int b)
{
    int gcd=1;
    for(int i=1;i<=a && i<=b;i++)
    {
        if(a%i==0 && b%i==0)
            gcd=i;
    }
    return gcd;
}

int findLCM(int a,int b)
{
    int gcd=findGCD(a,b);
    int lcm=(a*b)/gcd;
    return lcm;
}

int main()
{
    int a,b;
    printf("Enter Two numbers: ");
    scanf("%d%d",&a,&b);
    int gcd=findGCD(a,b);
    int lcm=findLCM(a,b);

    printf("GCD is %d\n",gcd);
    printf("LCM is %d\n",lcm);

    return 0;
}
```
### Output
```
Enter Two numbers: 5 10
GCD is 5
LCM is 10
```
