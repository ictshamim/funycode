## C Implementation
### 15. Write a C program to print all Armstrong numbers between given interval using functions.
```c
//Write a C program to print all Armstrong numbers between given interval using functions.
#include<stdio.h>

void Armstrong(int a,int b)
{
    for(int i=a;i<=b;i++)
    {
        int k=i;
        int count=0;
        while(k>0)
        {
            ++count;
            k/=10;
        }
        int number=0;
        k=i;
        while(k)
        {
            int x=k%10;
            int p=1;
            for (int i = 1; i <= count; ++i)
            {
                p*=x;
            }
            number+=p;
            k/=10;
        }
        if(i==number)
            printf("%d is Armstrong Number\n",i);
        else
            printf("%d is Not Armstrong Number\n",i);
    }
}


int main()
{
    int a,b;
    scanf("%d%d",&a,&b);

    Armstrong(a,b);

    return 0;
}
```

5 is Armstrong Number
